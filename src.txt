class Solution {
    public int coinChange(int[] coins, int amount) {
        // Edge case: if amount is 0, we need 0 coins
        if (amount == 0) return 0;

        // dp[i] = minimum number of coins needed to make sum i
        int[] dp = new int[amount + 1];

        // Initialize with a large value (amount + 1 is safe "infinity")
        for (int i = 1; i <= amount; i++) {
            dp[i] = amount + 1;
        }

        dp[0] = 0; // base case

        // For each coin, update the dp table
        for (int coin : coins) {
            for (int curr = coin; curr <= amount; curr++) {
                dp[curr] = Math.min(dp[curr], dp[curr - coin] + 1);
            }
        }

        // If dp[amount] is still "infinity", it's not possible
        return dp[amount] > amount ? -1 : dp[amount];
    }
}
